MariaDB [projects]> CREATE VIEW EmpProject (empNo, projNo, totalHours)
    -> AS
    -> SELECT
    ->     w.empNo,
    ->     w.projNo,
    ->     SUM(w.hoursWorked) AS totalHours
    -> FROM Employee e, Project p, WorksOn w
    -> WHERE e.empNo = w.empNo
    ->   AND p.projNo = w.projNo
